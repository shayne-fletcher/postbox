8/// A monotone cell (LVar): state only increases via lattice `join`.
9pub struct LVar<L>
10where
11    L: JoinSemilattice + Clone + Send + Sync + PartialEq + 'static,
12{
13    inner: Arc<Inner<L>>,
14}
15
16struct Inner<L>
17where
18    L: JoinSemilattice + Clone + Send + Sync + PartialEq + 'static,
19{
20    state: Mutex<L>,
21    tx: watch::Sender<L>,
22}
23
24impl<L> Clone for LVar<L>
25where
26    L: JoinSemilattice + Clone + Send + Sync + PartialEq + 'static,
27{
28    fn clone(&self) -> Self {
29        Self {
30            inner: Arc::clone(&self.inner),
31        }
32    }
33}
34
35impl<L> LVar<L>
36where
37    L: BoundedJoinSemilattice + Clone + Send + Sync + PartialEq + 'static,
38{
39    /// Create a cell initialized to ⊥.
40    pub fn new() -> Self {
41        let bottom = L::bottom();
42        let (tx, _rx) = watch::channel(bottom.clone());
43        Self {
44            inner: Arc::new(Inner {
45                state: Mutex::new(bottom),
46                tx,
47            }),
48        }
49    }
50
51    /// Read current value (clone).
52    pub fn get(&self) -> L {
53        self.inner.state.lock().unwrap().clone()
54    }
55
56    /// Monotone write: state := state ∨ delta. Returns true if it
57    /// changed.
58    pub fn put_join(&self, delta: &L) -> bool {
59        let mut g = self.inner.state.lock().unwrap();
60        let new_v = g.join(delta);
61        if *g != new_v {
62            *g = new_v.clone();
63            let _ = self.inner.tx.send(new_v);
64            true
65        } else {
66            false
67        }
68    }
69
70    /// Wait until `target ≤ current`. Returns the (≥ target) value.
71    pub async fn await_at_least(&self, target: &L) -> L {
72        // Fast path: read the current state directly
73        {
74            let cur = self.inner.state.lock().unwrap().clone();
75            if target.leq(&cur) {
76                return cur;
77            }
78        }
79
80        // Then subscribe for changes
81        let mut rx = self.inner.tx.subscribe();
82        loop {
83            if rx.changed().await.is_err() {
84                // sender dropped; return last seen value
85                return rx.borrow().clone();
86            }
87            let cur = rx.borrow().clone();
88            if target.leq(&cur) {
89                return cur;
90            }
91        }
92    }
93
94    pub async fn await_monotone<F>(&self, mut p: F) -> L
95    where
96        F: FnMut(&L) -> bool + Send,
97    {
98        // Fast path via mutex
99        {
100            let cur = self.inner.state.lock().unwrap().clone();
101            if p(&cur) {
102                return cur;
103            }
104        }
105
106        // Then subscribe
107        let mut rx = self.inner.tx.subscribe();
108        loop {
109            if rx.changed().await.is_err() {
110                return rx.borrow().clone();
111            }
112            let cur = rx.borrow().clone();
113            if p(&cur) {
114                return cur;
115            }
116        }
117    }
118}

120#[cfg(all(test, feature = "async"))]
121mod tests {
122    use super::*;
123    use crate::join_semilattice::Max;
124    use std::collections::HashSet;
125    use std::time::Duration;
126    use tokio::time::timeout;
127
128    fn set<T: Eq + std::hash::Hash + Clone>(xs: &[T]) -> HashSet<T> {
129        xs.iter().cloned().collect()
130    }
131
132    #[tokio::test]
133    async fn await_at_least_waits_for_union() {
134        let cell = LVar::<HashSet<&'static str>>::new();
135
136        let target = set(&["a", "b", "c"]);
137        let target_in_task = target.clone();
138        let waiter = {
139            let cell = cell.clone();
140            tokio::spawn(async move { cell.await_at_least(&target_in_task).await })
141        };
142
143        assert!(cell.put_join(&set(&["a"])));
144        assert!(cell.put_join(&set(&["c"])));
145        assert!(cell.put_join(&set(&["b"])));
146
147        let got = waiter.await.unwrap();
148        assert!(target.is_subset(&got));
149    }
150
151    #[tokio::test]
152    async fn await_monotone_counter_reaches_threshold() {
153        type C = Option<Max<i64>>;
154        let cell = LVar::<C>::new();
155
156        let done = {
157            let cell = cell.clone();
158            tokio::spawn(async move {
159                cell.await_monotone(|v| matches!(v, Some(Max(n)) if *n >= 10))
160                    .await
161            })
162        };
163
164        assert!(cell.put_join(&Some(Max(3))));
165        assert!(cell.put_join(&Some(Max(7))));
166        assert!(!cell.put_join(&Some(Max(7))));
167        assert!(cell.put_join(&Some(Max(12)))); // triggers
168
169        let v = done.await.unwrap();
170        assert!(matches!(v, Some(Max(n)) if n >= 10));
171    }
172
173    #[tokio::test]
174    async fn fast_path_immediate_resolution() {
175        let cell = LVar::<HashSet<i32>>::new();
176        cell.put_join(&set(&[1, 2, 3, 4]));
177
178        let target = set(&[2, 4]);
179        let got = timeout(Duration::from_millis(50), cell.await_at_least(&target))
180            .await
181            .expect("timed out");
182        assert!(target.is_subset(&got));
183    }
